273     * @return: The map of hexagonal neighbors
274     */
275    private Map createHexagonalNeighborsMap(int i, int j) {
276       Map<Integer, Cell> neighbors = new HashMap<Integer, Cell>();
277       int[] x = new int[]{-1, -1, 0, 0, 1, 1};
278       int[] y = new int[]{0, 1, 1, -1, -1, 0};
279       addCellsToMap(i, j, x, y, neighbors);
280       return neighbors;
281    }

283    /**
284     * Adds cells to a map if not out of bounds
285     * @param i: The x position of the particular cell
286     * @param j: The y position of the particular cell
287     * @param x: An array of x positions 
288     * @param y: An array of y positions
289     * @param map: The map that needs to be added to
290     */
291    private void addCellsToMap(int i, int j, int[] x, int[] y, Map map) {
292       for(int k = 0; k < x.length; k++) {
293          if (checkBounds(i + x[k], j + y[k])) {
294             map.put(cellMatrix[i + x[k]][j + y[k]].cellID, cellMatrix[i + x[k]][j + y[k]]);
295          }
296       }
297    }

300     * Checks to make sure referenced locations are in the bounds of the grid
301     * @param i: The x position of the particular location
302     * @param j: The x position of the particular location
303     * @return
304     */
305    private boolean checkBounds(int i, int j) {
306       return (i < ApplicationConstants.NUM_OF_COLUMNS && i >= 0 && j < ApplicationConstants.NUM_OF_ROWS && j >= 0);
307    }

309    /**
310     * Populates the neighbors of each of the cells in the cell matrix
311     */
312    public void populateMatrixNeighborMaps(CellXMLReader cxr) {
313       for(int j = 0; j < ApplicationConstants.NUM_OF_ROWS; j++) {
314          for(int i = 0; i < ApplicationConstants.NUM_OF_COLUMNS; i++) {
315             if(cxr.getGridEdgeType().equalsIgnoreCase("toroidal") && cxr.getGridNeighborType().equalsIgnoreCase("corner"))
316                cellMatrix[i][j].neighbors = createToroidalCornerNeighborsMap(i, j);
317             else if(cxr.getGridEdgeType().equalsIgnoreCase("toroidal") && cxr.getGridNeighborType().equalsIgnoreCase("cardinal"))
318                cellMatrix[i][j].neighbors = createToroidalCardinalNeighborsMap(i, j);
319             else if(cxr.getGridEdgeType().equalsIgnoreCase("toroidal") && cxr.getGridNeighborType().equalsIgnoreCase("square"))
320                cellMatrix[i][j].neighbors = createToroidalSquareNeighborsMap(i, j);
321             else if(cxr.getGridEdgeType().equalsIgnoreCase("finite") && cxr.getGridNeighborType().equalsIgnoreCase("corner"))
322                cellMatrix[i][j].neighbors = createCornerNeighborsMap(i, j);
323             else if(cxr.getGridEdgeType().equalsIgnoreCase("finite") && cxr.getGridNeighborType().equalsIgnoreCase("cardinal"))
324                cellMatrix[i][j].neighbors = createCardinalNeighborsMap(i, j);
325             else if(cxr.getGridEdgeType().equalsIgnoreCase("finite") && cxr.getGridNeighborType().equalsIgnoreCase("square"))
326                cellMatrix[i][j].neighbors = createSquareNeighborsMap(i, j);
327             else if(cxr.getGridLocationShape().equalsIgnoreCase("hexagonal"))
328                cellMatrix[i][j].neighbors = createHexagonalNeighborsMap(i, j);
329             else
330                cellMatrix[i][j].neighbors = createSquareNeighborsMap(i, j);
331          }   
332       }
